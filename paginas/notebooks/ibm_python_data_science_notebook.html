class Circle(object):
    
    def __init__(self, radius = 10, color = 'red'):
        self.radius = radius
        self.color = color
    
    def addRadius(self, radius):
        self.radius = self.radius + radius
        
    def changeColor(self, color):
        self.color = color
    
class Rectangle(object):
    
    def __init__(self, color, height, width):
        self.color = color
        self.height = height
        self.width = width
